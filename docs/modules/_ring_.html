getTupleRing

  • getTupleRing<T>(...rings: T): Ring<{}>
    • Defined in Dev/fp-ts-numerics/src/Ring.ts:88

    Given a tuple of Rings returns a Ring for the tuple

    example

    import { getTupleRing } from 'fp-ts-numerics/Ring' import { fieldNumber } from 'fp-ts-numerics/Field'

    const R = getTupleRing(fieldNumber, fieldNumber, fieldNumber) assert.deepStrictEqual(R.add([1, 2, 3], [4, 5, 6]), [5, 7, 9]) assert.deepStrictEqual(R.mul([1, 2, 3], [4, 5, 6]), [4, 10, 18]) assert.deepStrictEqual(R.one, [1, 1, 1]) assert.deepStrictEqual(R.sub([1, 2, 3], [4, 5, 6]), [-3, -3, -3]) assert.deepStrictEqual(R.zero, [0, 0, 0])
